title_id varchar PK

employees as emp
-
emp_no VARCHAR PK
emp_title_id varchar(15) FK - titles.title_id
birth_date varchar(10)
first_name varchar
last_name varchar
sex varchar(1)
hire_date varchar(10)

departments as dep
-
dept_no varchar(4) PK
dept_name varchar UNIQUE

dept_emp
-
emp_no int PK FK - emp.emp_no
dept_no varchar(4) PK FK - dep.dept_no

dept_manager
-
emp_no VARCHAR FK - emp.emp_no
dept_no varchar PK FK - dep.dept_no

salaries
-
emp_no int PK FK - emp.emp_no
